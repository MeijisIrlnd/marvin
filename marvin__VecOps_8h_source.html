28 template <NumericType T>
29 void add(T* lhs, const T* rhs, size_t size) noexcept {
30 const auto vectorisedLoopSize = (size / mipp::N<T>()) * mipp::N<T>();
31 mipp::Reg<T> rl, rr;
32 for (auto i = 0_sz; i < vectorisedLoopSize; i += mipp::N<T>()) {
33 rl.load(&lhs[i]);
34 rr.load(&rhs[i]);
35 rl += rr;
36 rl.store(&lhs[i]);
37 }
38 for (auto i = vectorisedLoopSize; i < size; ++i) {
39 lhs[i] += rhs[i];
40 }
41 }
