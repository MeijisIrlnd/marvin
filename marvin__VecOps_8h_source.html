25 template <NumericType T>
26 void add(T* lhs, const T* rhs, size_t size) noexcept {
27 const auto vectorisedLoopSize = (size / mipp::N<T>()) * mipp::N<T>();
28 mipp::Reg<T> rl, rr;
29 for (auto i = 0_sz; i < vectorisedLoopSize; i += mipp::N<T>()) {
30 rl.load(&lhs[i]);
31 rr.load(&rhs[i]);
32 rl += rr;
33 rl.store(&lhs[i]);
34 }
35 for (auto i = vectorisedLoopSize; i < size; ++i) {
36 lhs[i] += rhs[i];
37 }
38 }
