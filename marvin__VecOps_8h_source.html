27 template <NumericType T>
28 void add(T* lhs, const T* rhs, size_t size) noexcept {
29 const auto vectorisedLoopSize = (size / mipp::N<T>()) * mipp::N<T>();
30 mipp::Reg<T> rl, rr;
31 for (auto i = 0_sz; i < vectorisedLoopSize; i += mipp::N<T>()) {
32 rl.load(&lhs[i]);
33 rr.load(&rhs[i]);
34 rl += rr;
35 rl.store(&lhs[i]);
36 }
37 for (auto i = vectorisedLoopSize; i < size; ++i) {
38 lhs[i] += rhs[i];
39 }
40 }
