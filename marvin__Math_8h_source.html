53 template <FloatType T>
54 [[nodiscard]] T gainToDb(T gain, T minusInfDb = static_cast<T>(-100.0)) noexcept {
55 const auto clamped = std::clamp(gain, static_cast<T>(0.0), static_cast<T>(1.0));
56 if (clamped <= 0.0f) return minusInfDb;
57 const auto db = std::max(static_cast<T>(20.0) * std::log10(clamped), minusInfDb);
