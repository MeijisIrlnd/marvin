22 [[nodiscard]] BiquadCoefficients<SampleType> lowpass(double sampleRate, SampleType cutoff, SampleType q) noexcept {
23 constexpr static auto twoPi = std::numbers::pi_v<SampleType> * static_cast<SampleType>(2.0);
24 const auto fs{ static_cast<SampleType>(sampleRate) };
25 const auto omega{ twoPi * (cutoff / fs) };
26 const auto cosOmega = std::cos(omega);
27 const auto sinOmega = std::sin(omega);
28 const auto alpha{ sinOmega / (static_cast<SampleType>(2.0) * q) };
29 const auto b0 = (static_cast<SampleType>(1.0) - cosOmega) / static_cast<SampleType>(2.0);
30 const auto b1 = static_cast<SampleType>(1.0) - cosOmega;
31 const auto b2 = b0;
32 const auto a0 = static_cast<SampleType>(1.0) + alpha;
33 const auto a1 = static_cast<SampleType>(-2.0) * cosOmega;
34 const auto a2 = static_cast<SampleType>(1.0) - alpha;
35 return {
36 .b0 = b0,
37 .b1 = b1,
38 .b2 = b2,
39 .a0 = a0,
40 .a1 = a1,
